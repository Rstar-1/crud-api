const Dashboard = () => {
  const [activeIndex, setActiveIndex] = useState(0);
  const containerRef = useRef(null);
  const wrapperRefs = useRef([]);

  useEffect(() => {
    // Inject Plus Jakarta Sans font dynamically for premium typography
    const link = document.createElement('link');
    link.href = 'https://fonts.googleapis.com/css2?family=Plus+Jakarta+Sans:wght@300;400;500;600;700;800;900&display=swap';
    link.rel = 'stylesheet';
    document.head.appendChild(link);

    const handleScroll = () => {
      const newProgresses = [];
      const stickyTop = window.innerHeight * 0.2; // 20vh sticky offset

      wrapperRefs.current.forEach((wrapper, index) => {
        if (!wrapper) return;
        const rect = wrapper.getBoundingClientRect();

        // Active row height when expanded is 360px
        const rowHeight = 360;
        const distance = rect.height - rowHeight - stickyTop;
        const offset = stickyTop - rect.top;

        let progress = distance > 0 ? offset / distance : 0;
        progress = Math.min(Math.max(progress, 0), 1);

        newProgresses[index] = progress;

        const titleEl = document.getElementById(`title-${index}`);
        if (titleEl) {
          titleEl.style.setProperty('--bg-pos', `${(1 - progress) * 100}% 0%`);
        }
      });

      // Find the active index based on progresses
      let newActiveIndex = 0;
      for (let i = 0; i < items.length; i++) {
        const prog = newProgresses[i] || 0;
        if (prog > 0 && prog < 1) {
          newActiveIndex = i;
          break;
        }
        if (prog === 1) {
          newActiveIndex = i;
        }
      }

      setActiveIndex(newActiveIndex);
    };

    window.addEventListener('scroll', handleScroll, { passive: true });
    window.addEventListener('resize', handleScroll);

    // Initial run
    handleScroll();

    return () => {
      window.removeEventListener('scroll', handleScroll);
      window.removeEventListener('resize', handleScroll);
      if (document.head.contains(link)) {
        document.head.removeChild(link);
      }
    };
  }, []);

  const handleRowClick = (index) => {
    const wrapper = document.getElementById(`wrapper-${index}`);
    if (wrapper) {
      const rect = wrapper.getBoundingClientRect();
      const scrollTop = window.scrollY || document.documentElement.scrollTop;
      const targetY = rect.top + scrollTop - window.innerHeight * 0.2;
      window.scrollTo({ top: targetY, behavior: 'smooth' });

                const viewCenter = window.innerWidth / 2;
                const distanceFromCenter = Math.abs(slideCenter - viewCenter);

                // 1 when perfectly centered, 0 when off-screen
                const factor = Math.max(0, 1 - (distanceFromCenter / (window.innerWidth * 0.8)));
                return factor;
            });

            setSlideVisibilities(visibilities);
        };

        window.addEventListener('scroll', handleScroll, { passive: true });
        handleScroll(); // Initial run

                            <span>{errorMsg}</span>
                            <button
                                style={{ background: 'none', border: 'none', color: '#fca5a5', cursor: 'pointer', fontWeight: 'bold' }}
                                onClick={fetchStats}
                            >
                                Retry
                            </button>
                        </div>
                    )}

                    <form onSubmit={handleSend} className="input-row">
                        <textarea
                            className="chat-input"
                            value={input}
                            onChange={(e) => setInput(e.target.value)}
                            onKeyDown={handleKeyDown}
                            placeholder={activeMode === 'image' ? "Describe the image you want to generate..." : "Type a message or prompt..."}
                            rows="1"
                            disabled={loading}
                        />
                        <button type="submit" className="send-btn" disabled={loading || !input.trim()}>
                            {loading ? 'Processing...' : (activeMode === 'image' ? 'Generate' : 'Send')}
                        </button>
                    </form>
                </div>
            </main>
        </div>
    );
};

import React, { useEffect, useRef } from 'react';

const SmoothScroll = ({ children }) => {
    const targetY = useRef(0);
    const currentY = useRef(0);
    const ease = 0.08; // Adjust between 0.05 and 0.12 to fine-tune glide inertia (lower = slower/smoother)

    useEffect(() => {
        // Initialize position to current scroll position
        targetY.current = window.scrollY;
        currentY.current = window.scrollY;

        const handleWheel = (e) => {
            // Prevent browser's instant scroll behavior
            e.preventDefault();

            // Calculate current maximum scrollable height
            const maxScroll = document.documentElement.scrollHeight - window.innerHeight;

            // Increment target scroll position by wheel delta
            targetY.current = Math.max(0, Math.min(maxScroll, targetY.current + e.deltaY));
        };

        const handleScroll = () => {
            // Sync variables if scrolling was triggered by native actions (scrollbar drag, keyboard, hash link)
            if (Math.abs(window.scrollY - currentY.current) > 1.5) {
                targetY.current = window.scrollY;
                currentY.current = window.scrollY;
            }
        };

        // Bind wheel event as non-passive to allow e.preventDefault()
        window.addEventListener('wheel', handleWheel, { passive: false });
        window.addEventListener('scroll', handleScroll, { passive: true });

        let rafId;
        const tick = () => {
            // Linear interpolation (lerp) for smooth glide
            const diff = targetY.current - currentY.current;
            currentY.current += diff * ease;

            if (Math.abs(diff) < 0.05) {
                currentY.current = targetY.current;
            } else {
                // Scroll window natively to the smoothed position
                window.scrollTo(0, currentY.current);
            }

            rafId = requestAnimationFrame(tick);
        };

        tick();

        return () => {
            cancelAnimationFrame(rafId);
            window.removeEventListener('wheel', handleWheel);
            window.removeEventListener('scroll', handleScroll);
        };
    }, []);

    return <>{children}</>;
};

export default SmoothScroll;
