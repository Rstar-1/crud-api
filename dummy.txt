const DataTable = ({
    title,
    subtitle,
    data = [],
    columns = [],
    totalItems = 0,
    itemsPerPage = 10,
    page = 1,
    onPageChange,
    searchQuery = "",
    onSearchChange,
    searchPlaceholder = "Search...",
    showControls = true,
    itemName = "items",
    loading = false
}) => {
    return (
        <>
            {(title || (showControls && onSearchChange)) && (
                <div className="flex items-center">
                    {title && (
                        <div className="w-70">
                            <h2 className="mid-text text-dark font-500">
                                {title}
                            </h2>
                            {subtitle && <p className="text-gray font-400 mini-text mt-1">{subtitle}</p>}
                        </div>
                    )}
                    {showControls && onSearchChange && (
                        <div className="w-30">
                            <div className="relative w-full overflow-hidden rounded-5 border-ec">
                                <input
                                    type="text"
                                    placeholder={searchPlaceholder}
                                    className="text-dark bg-white h-input mini-text border-0 w-full"
                                    value={searchQuery}
                                    onChange={(e) => onSearchChange(e.target.value)}
                                />
                                <span className="absolute top-0 right-0 bg-white py-10 px-14 text-gray">
                                    <svg width="16" height="16" fill="none" stroke="currentColor" strokeWidth="2.5" viewBox="0 0 24 24">

                    <thead>
                        <tr>
                            {columns.map((col, idx) => (
                                <th
                                    key={idx}
                                    style={{
                                        ...col.style
                                    }}

                            ))}
                        </tr>
                    </thead>
                    <tbody>
                        {loading ? (
                            <tr>

                                    </div>
                                </td>
                            </tr>

                            data.map((row, rowIdx) => (
                                <tr key={row._id || row.id || rowIdx}>
                                    {columns.map((col, colIdx) => {
                                        const cellValue = col.accessor ? row[col.accessor] : undefined;

                                                                {val}
                                                            </p>
                                                        ))}
                                                    </div>
                                                );
                                            } else if (col.ui === "status") {

                                                let badgeBg = "#f1f5f9";

                                                    </span>

                                                                <rect x="3" y="11" width="18" height="11" rx="2" ry="2"></rect>
                                                                <path d="M7 11V7a5 5 0 0 1 9.9-1"></path>
                                                            </svg>
                                                        )}
                                                        {cellValue}
                                                    </span>
                                                );
                                            } else if (col.ui === "text") {

                                                cellContent = (
                                                    <p className="text-gray line-clamp1 mini-text" title={cellValue || ""} style={{ margin: 0 }}>
                                                        {cellValue || "-"}
                                                    </p>
                                                );
                                            }
                                        }
                                        return (
                                            <td
                                                key={colIdx}
                                                style={{
                                                    verticalAlign: "middle",
                                                    ...col.style
                                                }}
                                                className={`p-12 text-dark bordb ${col.className || ""}`}
                                            >
                                                {cellContent}
                                            </td>
                                        );
                                    })}
                                </tr>
                            ))
                        )}
                    </tbody>
                </table>
            </div>

            {onPageChange && totalItems > itemsPerPage && (
                <Pagination
                    page={page}
                    totalItems={totalItems}
                    itemsPerPage={itemsPerPage}
                    onPageChange={onPageChange}
                    itemName={itemName}
                />
            )}
        </>
    );
};

                                        const char = e.target.value.slice(-1);
                                        if (char && !/^\d+$/.test(char)) return;
                                        const newOtpArray = [...otpArray];
                                        newOtpArray[index] = char;
                                        const newVal = newOtpArray.join("");
                                        onChange?.(newVal);

                                        // Auto-focus next input if we typed a character
                                        if (char && index < count - 1) {
                                            otpRefs.current[index + 1]?.focus();
                                        }
                                    }}
                                    onKeyDown={(e) => {
                                        if (e.key === "Backspace") {
                                            const newOtpArray = [...otpArray];
                                            if (e.target.value) {
                                                newOtpArray[index] = "";
                                                onChange?.(newOtpArray.join(""));
                                            } else if (index > 0) {
                                                newOtpArray[index - 1] = "";
                                                onChange?.(newOtpArray.join(""));
                                                otpRefs.current[index - 1]?.focus();
                                            }
                                        }
                                    }}
                                    className="text-center font-600 text-gray bg-white border-0 rounded-5 mini-text"
                                    style={{
                                        width: "42px",
                                        height: "42px",
                                        outline: "none",
                                        transition: "all 0.2s",
                                    }}
                                    onFocus={(e) => e.target.style.borderColor = "var(--primary)"}
                                    onBlur={(e) => e.target.style.borderColor = "var(--gray)"}
                                />
                            );
                        })}
                    </div>
                );
            }

            default:
                return null;
        }
    };

    return (
        <div
            className="w-full grid-cols-1 gap-6"
        >
            {label && (
                <label className="mini-text font-500" style={{ fontSize: "14px", color: "#4b5563" }}>
                    {label}
                </label>
            )}
            {renderField()}
            {error && (
                <small className="text-danger mt-2 mini-text">
                    {error}
                </small>
            )}
        </div>
    );
};

import React, { useMemo, useState } from "react";
import Fields from "./Fields";

const FormBuilder = ({ fields = [], onSubmit, submitType = "formdata", col = "1" }) => {
    const getDefaultValue = (field) => {
        if (field.defaultValue !== undefined) return field.defaultValue;
        if (field.type === "range-datepicker") return { fromDate: "", toDate: "" };
        if (field.type === "multiselect") return [];
        if (field.type === "quantity") return 1;
        if (field.type === "rating") return 0;
        if (field.type === "checkbox" || field.type === "radio" || field.type === "switch") return false;
        return "";
    };

    const initialValues = useMemo(() => {
        return fields.reduce((acc, field) => {
            acc[field.name] = getDefaultValue(field);
            return acc;
        }, {});
    }, [fields]);

    const [form, setForm] = useState(initialValues);

    const handleChange = (name, value) => {
        setForm((prev) => ({
            ...prev,
            [name]: value,
        }));
    };

    const handleSubmit = () => {
        if (submitType === "json") {
            onSubmit?.(form);
            return;
        }

        const formData = new FormData();
        Object.entries(form).forEach(([key, val]) => {
            if (val === null || val === undefined) return;
            if (val instanceof FileList) {
                Array.from(val).forEach((file) => {
                    formData.append(key, file);
                });
            } else if (Array.isArray(val)) {
                val.forEach((item) => {
                    if (item instanceof File) {
                        formData.append(key, item);
                    } else {
                        formData.append(key, item);
                    }
                });
            } else if (val instanceof File) {
                formData.append(key, val);
            } else if (typeof val === "object") {
                formData.append(key, JSON.stringify(val));
            } else {
                formData.append(key, val);
            }
        });
        onSubmit?.(formData);
    };

    return (
        <form
            onSubmit={(e) => {
                e.preventDefault();
                handleSubmit();
            }}
            className="w-full"
        >
            <div className={`grid-cols-${col} gap-12 items-start`}>
                {fields.map((field) => (
                    <Fields
                        key={field.name}
                        {...field}
                        value={form[field.name]}
                        onChange={(value) => handleChange(field.name, value)}
                    />
                ))}
            </div>

            <div className="flex justify-center mt-20">
                <button
                    type="submit"
                    className="bg-primary mini-text text-white px-12 py-8 rounded-5"
                >
                    Save Changes
                </button>
            </div>
        </form>
    );
};

export default FormBuilder;

const Modal = ({

    footer,

    closeOnOverlayClick = true,
    type = "modal",

}) => {
    const [localIsOpen, setLocalIsOpen] = useState(false);

    const isControlled = controlledIsOpen !== undefined;
    const isOpen = isControlled ? controlledIsOpen : localIsOpen;

    const handleClose = () => {
        if (isControlled) {
            controlledOnClose?.();
        } else {
            setLocalIsOpen(false);
        }
    };

    const handleOpen = () => {
        if (!isControlled) {
            setLocalIsOpen(true);
        }
    };

    const getSizeStyle = () => {
        switch (size) {

                return { width: "30%" };
            case "md":
                return { width: "40%" };
            case "lg":

            case "xl":
                return { width: "60%" };

                return { width: "80%" };
            default:

            {trigger && React.cloneElement(trigger, {

                    className={wrapperClass}

                            backgroundColor: "rgba(15, 22, 35, 0.55)",

                            WebkitBackdropFilter: "blur(6px)",

                        className={cardClass}

                    >
                        {/* Modal Header */}

                                className="mid-text font-500 text-dark"
                            >
                                {title || "Modal Title"}
                            </h3>

                                <svg
                                    viewBox="0 0 24 24"
                                    width="20"

                                >

                            className="overflow-auto w-full"
                            style={isSidebar ? { height: "auto" } : { height: bodyHeight || "auto", maxHeight: bodyHeight ? undefined : "400px" }}
                        >
                            <div className="px-20 py-10">
                                {children}

                            {/* Modal Footer */}
                            {footer !== null && (
                                <div
                                    className="flex items-center justify-center gap-4 bordh p-14"
                                >
                                    {footer && footer !== true ? (

                                                onClick={handleClose}

                                            </button>

Pagination
import React from "react";

const Pagination = ({
    page,
    totalItems,
    itemsPerPage,
    onPageChange,
    itemName = "items"
}) => {
    const totalPages = Math.ceil(totalItems / itemsPerPage) || 1;
    const startItem = totalItems === 0 ? 0 : (page - 1) * itemsPerPage + 1;
    const endItem = Math.min(page * itemsPerPage, totalItems);

    if (totalItems === 0) return null;

    const handlePageClick = (pNum) => {
        if (pNum >= 1 && pNum <= totalPages && pNum !== page) {
            onPageChange(pNum);
        }
    };

    const btnStyle = {
        width: "28px",
        height: "28px",
        fontSize: "0.775rem",
        border: "1px solid #ececec",
        display: "flex",
        alignItems: "center",
        justifyContent: "center",
        cursor: "pointer",
        transition: "all 0.2s ease",
        borderRadius: "4px"
    };

    return (
        <div className="bg-tertiary mt-4 w-full rounded-5">
            <div className="flex items-center justify-between p-10">
                <p className="text-gray mini-text">
                    Showing {startItem} to {endItem} of {totalItems} {itemName}
                </p>
                <div className="flex items-center gap-4">
                    <button
                        className="bg-white text-gray font-500"
                        style={{
                            ...btnStyle,
                            opacity: page === 1 ? 0.4 : 1,
                            cursor: page === 1 ? "not-allowed" : "pointer"
                        }}
                        disabled={page === 1}
                        onClick={() => handlePageClick(1)}
                        title="First Page"
                    >
                        &laquo;
                    </button>
                    <button
                        className="bg-white text-gray font-500"
                        style={{
                            ...btnStyle,
                            opacity: page === 1 ? 0.4 : 1,
                            cursor: page === 1 ? "not-allowed" : "pointer"
                        }}
                        disabled={page === 1}
                        onClick={() => handlePageClick(page - 1)}
                        title="Previous Page"
                    >
                        &lt;
                    </button>
                    {Array.from({ length: totalPages }).map((_, i) => {
                        const pNum = i + 1;
                        // Show current page, and a few neighboring pages
                        if (totalPages > 5 && Math.abs(pNum - page) > 2) {
                            if (pNum === 1 || pNum === totalPages) {
                                return (
                                    <span key={pNum} className="text-gray px-4" style={{ fontSize: "0.775rem" }}>
                                        ...
                                    </span>
                                );
                            }
                            return null;
                        }
                        const isActive = page === pNum;
                        return (
                            <button
                                key={pNum}
                                className={isActive ? "bg-primary text-white font-500" : "bg-white text-gray font-500"}
                                style={{
                                    ...btnStyle,
                                    color: isActive ? "#ffffff" : "var(--gray)",
                                    backgroundColor: isActive ? "var(--primary)" : "#ffffff",
                                    borderColor: isActive ? "var(--primary)" : "#ececec"
                                }}
                                onClick={() => handlePageClick(pNum)}
                            >
                                {pNum}
                            </button>
                        );
                    })}
                    <button
                        className="bg-white text-gray font-500"
                        style={{
                            ...btnStyle,
                            opacity: page === totalPages ? 0.4 : 1,
                            cursor: page === totalPages ? "not-allowed" : "pointer"
                        }}
                        disabled={page === totalPages}
                        onClick={() => handlePageClick(page + 1)}
                        title="Next Page"
                    >
                        &gt;
                    </button>
                    <button
                        className="bg-white text-gray font-500"
                        style={{
                            ...btnStyle,
                            opacity: page === totalPages ? 0.4 : 1,
                            cursor: page === totalPages ? "not-allowed" : "pointer"
                        }}
                        disabled={page === totalPages}
                        onClick={() => handlePageClick(totalPages)}
                        title="Last Page"
                    >
                        &raquo;
                    </button>
                </div>
            </div>
        </div>
    );
};

export default Pagination;

          >
            <div
              className="flex justify-between items-center p-12"
              style={{
                borderBottom: '1px solid var(--border-color)',
                background: 'rgba(0, 0, 0, 0.02)'
              }}
            >
              <div className="flex gap-6">
                <button
                  onClick={() => setEditorTab("editor")}
                  style={{
                    background: editorTab === "editor" ? 'var(--bg-surface-hover)' : 'none',
                    border: 'none',
                    padding: '0.4rem 0.85rem',
                    borderRadius: '6px',
                    fontSize: '0.85rem',
                    fontWeight: 600,
                    color: editorTab === "editor" ? '#7a1be7' : 'var(--text-muted)',
                    cursor: 'pointer',
                    transition: 'all 0.2s'
                  }}
                >
                  JSON Editor
                </button>
                <button
                  onClick={() => setEditorTab("tree")}
                  style={{
                    background: editorTab === "tree" ? 'var(--bg-surface-hover)' : 'none',
                    border: 'none',
                    padding: '0.4rem 0.85rem',
                    borderRadius: '6px',
                    fontSize: '0.85rem',
                    fontWeight: 600,
                    color: editorTab === "tree" ? '#7a1be7' : 'var(--text-muted)',
                    cursor: 'pointer',
                    transition: 'all 0.2s'
                  }}
                >
                  Tree View
                </button>
              </div>
              <div className="flex gap-6">
                <button
                  onClick={handleFormat}
                  style={{
                    background: 'var(--bg-surface)',
                    border: '1px solid var(--border-color)',
                    padding: '0.4rem 0.85rem',
                    borderRadius: '6px',
                    fontSize: '0.85rem',
                    fontWeight: 500,
                    color: 'var(--text-secondary)',
                    cursor: 'pointer',
                    display: 'flex',
                    alignItems: 'center',
                    gap: '0.35rem',
                    transition: 'all 0.2s'
                  }}
                >
                  <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5">
                    <polyline points="4 17 10 11 4 5"></polyline>
                    <line x1="12" y1="19" x2="20" y2="19"></line>
                  </svg>
                  Format
                </button>
                <button
                  onClick={() => handleCopyText(jsonText, "JSON Copied!")}
                  style={{
                    background: 'var(--bg-surface)',
                    border: '1px solid var(--border-color)',
                    padding: '0.4rem 0.85rem',
                    borderRadius: '6px',
                    fontSize: '0.85rem',
                    fontWeight: 500,
                    color: 'var(--text-secondary)',
                    cursor: 'pointer',
                    display: 'flex',
                    alignItems: 'center',
                    gap: '0.35rem',
                    transition: 'all 0.2s'
                  }}
                >
                  <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5">
                    <rect x="9" y="9" width="13" height="13" rx="2" ry="2"></rect>
                    <path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1"></path>
                  </svg>
                  Copy
                </button>
              </div>
            </div>

            {/* Code / Tree content area */}
            {editorTab === "editor" ? (
              <div
                className="flex relative"
                style={{ flex: 1, overflow: 'hidden', background: '#0f172a', color: '#38bdf8' }}
              >
                <div
                  ref={lineNumbersRef}
                  style={{
                    padding: '1rem 0.5rem 1rem 1rem',
                    textAlign: 'right',
                    color: '#475569',
                    fontFamily: 'monospace',
                    fontSize: '0.9rem',
                    userSelect: 'none',
                    background: '#0b0f19',
                    borderRight: '1px solid #1e293b',
                    minWidth: '40px',
                    overflowY: 'hidden'
                  }}
                >
                  {getLineNumbers().map((num) => (
                    <div key={num}>{num}</div>
                  ))}
                </div>
                <textarea
                  ref={textareaRef}
                  value={jsonText}
                  onChange={(e) => handleJsonTextChange(e.target.value)}
                  onKeyUp={handleCursorMove}
                  onClick={handleCursorMove}
                  onScroll={handleScroll}
                  spellCheck="false"
                  style={{
                    flex: 1,
                    border: 'none',
                    background: 'transparent',
                    color: '#e2e8f0',
                    fontFamily: 'monospace',
                    fontSize: '0.9rem',
                    padding: '1rem',
                    outline: 'none',
                    resize: 'none',
                    overflowY: 'auto',
                    lineHeight: 1.5,
                    whiteSpace: 'pre'
                  }}
                />
              </div>
            ) : (
              <div
                style={{
                  flex: 1,
                  padding: '20px',
                  background: '#0f172a',
                  color: '#e2e8f0',
                  fontFamily: 'monospace',
                  fontSize: '0.95rem',
                  overflowY: 'auto'
                }}
              >
                {(() => {
                  try {
                    return renderTreeView(JSON.parse(jsonText));
                  } catch (e) {
                    return <span style={{ color: "#ef4444" }}>Invalid JSON - unable to render tree view.</span>;
                  }
                })()}
              </div>
            )}

            {/* Editor Footer */}
            <div
              className="flex justify-between items-center p-8"
              style={{
                borderTop: '1px solid var(--border-color)',
                fontSize: '0.75rem',
                color: 'var(--text-muted)',
                background: 'rgba(0, 0, 0, 0.02)'
              }}
            >
              <div
                style={{
                  color: isValid ? '#10b981' : '#ef4444',
                  fontWeight: 600,
                  display: 'flex',
                  alignItems: 'center',
                  gap: '0.35rem'
                }}
              >
                {isValid ? "✔ Valid JSON" : "✖ Invalid JSON"}
              </div>
              {editorTab === "editor" && (
                <div>
                  Ln {cursorPos.line}, Col {cursorPos.col} &nbsp;&nbsp;|&nbsp;&nbsp; Length: {jsonText.length} bytes
                </div>
              )}
            </div>
          </div>

          {/* Validation card & Quick Actions */}
          <div className="grid-cols-2 gap-8">
            <div
              className="p-12 border-ec rounded-10"
              style={{ background: "var(--bg-surface)" }}
            >
              <div className="flex items-center gap-6 mb-8 text-dark font-600 mini-text">
                <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5">
                  <path d="M22 11.08V12a10 10 0 1 1-5.93-9.14"></path>
                  <polyline points="22 4 12 14.01 9 11.01"></polyline>
                </svg>
                Validation Status
              </div>
              {isValid ? (
                <p className="mini-text text-success font-500">JSON syntax is perfectly correct.</p>
              ) : (
                <p className="mini-text text-danger font-500" style={{ fontFamily: "monospace", wordBreak: "break-all" }}>
                  {validationError}
                </p>
              )}
            </div>

            <div
              className="p-12 border-ec rounded-10"
              style={{ background: "var(--bg-surface)" }}
            >
              <div className="flex items-center gap-6 mb-8 text-dark font-600 mini-text">
                <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5">
                  <polygon points="12 2 2 7 12 12 22 7 12 2 12 2"></polygon>
                  <polyline points="2 17 12 22 22 17"></polyline>
                  <polyline points="2 12 12 17 22 12"></polyline>
                </svg>
                Quick Actions
              </div>
              <div className="flex gap-10 flex-wrap">
                <button
                  onClick={() => handleJsonTextChange(activeConfigTab === "modal" ? "{}" : "[]")}
                  className="bg-light-danger text-danger px-10 py-5 rounded-5 mini-text font-600 cursor-pointer"
                  style={{ border: "none" }}
                >
                  Clear All
                </button>
                <button
                  onClick={handleMinify}
                  className="bg-light-secondary text-secondary px-10 py-5 rounded-5 mini-text font-600 cursor-pointer"
                  style={{ border: "none" }}
                >
                  Minify JSON
                </button>
              </div>
            </div>
          </div>
        </div>

        {/* Right Column: Configurator Panel */}
        <div
          className="flex flex-column"
          style={{
            flex: 5,
            background: 'var(--bg-surface)',
            border: '1px solid var(--border-color)',
            borderRadius: '12px',
            boxShadow: 'var(--shadow-sm)',
            height: '100%',
            overflow: 'hidden'
          }}
        >
          {/* Tabs header */}
          <div
            className="flex"
            style={{ borderBottom: '1px solid var(--border-color)', background: 'rgba(0, 0, 0, 0.01)' }}
          >
            <button
              onClick={() => handleConfigTabChange("modal")}
              style={{
                background: 'none',
                border: 'none',
                padding: '0.9rem 1.25rem',
                color: activeConfigTab === "modal" ? '#7a1be7' : 'var(--text-muted)',
                fontSize: '0.85rem',
                fontWeight: 600,
                cursor: 'pointer',
                borderBottom: activeConfigTab === "modal" ? '2.5px solid #7a1be7' : 'none'
              }}
            >
              Modal
            </button>
            <button
              onClick={() => handleConfigTabChange("table")}
              style={{
                background: 'none',
                border: 'none',
                padding: '0.9rem 1.25rem',
                color: activeConfigTab === "table" ? '#7a1be7' : 'var(--text-muted)',
                fontSize: '0.85rem',
                fontWeight: 600,
                cursor: 'pointer',
                borderBottom: activeConfigTab === "table" ? '2.5px solid #7a1be7' : 'none'
              }}
            >
              Table
            </button>
            <button
              onClick={() => handleConfigTabChange("field")}
              style={{
                background: 'none',
                border: 'none',
                padding: '0.9rem 1.25rem',
                color: activeConfigTab === "field" ? '#7a1be7' : 'var(--text-muted)',
                fontSize: '0.85rem',
                fontWeight: 600,
                cursor: 'pointer',
                borderBottom: activeConfigTab === "field" ? '2.5px solid #7a1be7' : 'none'
              }}
            >
              Field
            </button>
          </div>

          {/* Configurator Scrollable Panel */}
          <div style={{ flex: 1, overflowY: 'auto', padding: '16px' }} className="flex flex-column gap-16">
            
            {/* 1. Modal Config UI */}
            {activeConfigTab === "modal" && (
              <div className="flex flex-column gap-16">
                <h4 className="mid-text font-600 text-dark">Modal Properties</h4>
                
                <div className="grid-cols-2 gap-12">
                  <Fields
                    type="input"
                    label="Modal Title"
                    value={modalConfig.title}
                    onChange={(val) => handleModalChange("title", val)}
                    border={true}
                  />
                  <Fields
                    type="select"
                    label="Modal Size"
                    value={modalConfig.size}
                    options={[
                      { label: "Small (sm)", value: "sm" },
                      { label: "Medium (md)", value: "md" },
                      { label: "Large (lg)", value: "lg" },
                      { label: "Extra Large (xl)", value: "xl" },
                      { label: "Full (full)", value: "full" }
                    ]}
                    onChange={(val) => handleModalChange("size", val)}
                    border={true}
                  />
                  <Fields
                    type="select"
                    label="Modal Type"
                    value={modalConfig.type}
                    options={[
                      { label: "Standard Modal", value: "modal" },
                      { label: "Sidebar Drawer", value: "sidebar" }
                    ]}
                    onChange={(val) => handleModalChange("type", val)}
                    border={true}
                  />
                  {modalConfig.type === "sidebar" && (
                    <Fields
                      type="select"
                      label="Drawer Placement"
                      value={modalConfig.placement}
                      options={[
                        { label: "Right Slide-in", value: "right" },
                        { label: "Left Slide-in", value: "left" }
                      ]}
                      onChange={(val) => handleModalChange("placement", val)}
                      border={true}
                    />
                  )}
                </div>

                <hr style={{ border: 'none', borderTop: '1px solid var(--border-color)', margin: '8px 0' }} />

                <div className="flex justify-between items-center">
                  <h5 className="small-text font-600 text-dark">Modal Form Fields</h5>
                  <button
                    onClick={handleAddModalField}
                    className="bg-light-primary text-primary px-10 py-5 rounded-5 mini-text font-600 cursor-pointer"
                    style={{ border: "none" }}
                  >
                    + Add Field
                  </button>
                </div>

                <div className="flex flex-column gap-10">
                  {(modalConfig.fields || []).map((field, idx) => (
                    <div
                      key={idx}
                      className="flex items-center gap-10 p-8 border-ec rounded-5"
                      style={{ background: "var(--bg-color)" }}
                    >
                      <div style={{ flex: 1 }}>
                        <Fields
                          type="input"
                          value={field.name}
                          placeholder="Field name (key)"
                          onChange={(val) => handleModalFieldChange(idx, "name", val)}
                          border={true}
                        />
                      </div>
                      <div style={{ flex: 1 }}>
                        <Fields
                          type="input"
                          value={field.label}
                          placeholder="Label text"
                          onChange={(val) => handleModalFieldChange(idx, "label", val)}
                          border={true}
                        />
                      </div>
                      <div style={{ width: "120px" }}>
                        <Fields
                          type="select"
                          value={field.type}
                          options={["input", "number", "email", "tel", "password", "datepicker", "color", "textarea", "switch"]}
                          onChange={(val) => handleModalFieldChange(idx, "type", val)}
                          border={true}
                        />
                      </div>
                      <button
                        onClick={() => handleDeleteModalField(idx)}
                        className="text-danger flex items-center justify-center p-4 cursor-pointer"
                        style={{ background: "none", border: "none" }}
                        title="Delete field"
                      >
                        🗑️
                      </button>
                    </div>
                  ))}
                  {(modalConfig.fields || []).length === 0 && (
                    <p className="mini-text text-gray text-center py-10">No fields configured for this Modal form.</p>
                  )}
                </div>

                {/* Code generation preview */}
                <div className="p-12 border-ec rounded-8" style={{ background: "#0b0f19", borderLeft: "3.5px solid #7a1be7" }}>
                  <div className="flex justify-between items-center mb-6">
                    <p className="mini-text font-600" style={{ color: "#7a1be7" }}>React Code Component</p>
                    <button
                      onClick={() => handleCopyText(getModalCode(), "React Modal Component Copied!")}
                      className="bg-light-primary text-primary px-8 py-3 rounded-5 mini-text cursor-pointer font-600"
                      style={{ border: "none" }}
                    >
                      Copy
                    </button>
                  </div>
                  <pre className="p-8 rounded-5 mini-text text-light" style={{ overflowX: "auto", fontFamily: "monospace", color: "#e2e8f0", background: "#0f172a" }}>
                    {getModalCode()}
                  </pre>
                </div>
              </div>
            )}

            {/* 2. Table Column Config UI */}
            {activeConfigTab === "table" && (
              <div className="flex flex-column gap-16">
                <div className="flex justify-between items-center">
                  <h4 className="mid-text font-600 text-dark">Table Column Setup</h4>
                  <div className="flex gap-8">
                    <button
                      onClick={() => {
                        setColumnsList(PRESETS.table);
                        setJsonText(JSON.stringify(PRESETS.table, null, 2));
                      }}
                      className="bg-light-secondary text-secondary px-8 py-4 rounded-5 mini-text font-500 cursor-pointer"
                      style={{ border: "none" }}
                    >
                      Load userFields Preset
                    </button>
                    <button
                      onClick={handleAddTableCol}
                      className="bg-light-primary text-primary px-10 py-4 rounded-5 mini-text font-600 cursor-pointer"
                      style={{ border: "none" }}
                    >
                      + Add Column
                    </button>
                  </div>
                </div>

                <div className="flex flex-column gap-10">
                  {columnsList.map((col, idx) => (
                    <div
                      key={idx}
                      className="flex items-center gap-10 p-8 border-ec rounded-5"
                      style={{ background: "var(--bg-color)" }}
                    >
                      <div style={{ flex: 1.2 }}>
                        <Fields
                          type="input"
                          value={col.header}
                          placeholder="Header Text"
                          onChange={(val) => handleTableColChange(idx, "header", val)}
                          border={true}
                        />
                      </div>
                      <div style={{ flex: 1 }}>
                        <Fields
                          type="input"
                          value={col.accessor}
                          placeholder="Accessor Key"
                          onChange={(val) => handleTableColChange(idx, "accessor", val)}
                          border={true}
                        />
                      </div>
                      <div style={{ width: "130px" }}>
                        <Fields
                          type="select"
                          value={col.ui || "text"}
                          options={[
                            { label: "Profile UI", value: "profile" },
                            { label: "Array Badges", value: "arr-badge" },
                            { label: "Status Dot", value: "status" },
                            { label: "Pill Badge", value: "badge" },
                            { label: "Icon Pill", value: "icon-badge" },
                            { label: "Plain Text", value: "text" },
                            { label: "Truncated Description", value: "desc" }
                          ]}
                          onChange={(val) => handleTableColChange(idx, "ui", val)}
                          border={true}
                        />
                      </div>
                      <button
                        onClick={() => handleDeleteTableCol(idx)}
                        className="text-danger flex items-center justify-center p-4 cursor-pointer"
                        style={{ background: "none", border: "none" }}
                        title="Delete column"
                      >
                        🗑️
                      </button>
                    </div>
                  ))}
                  {columnsList.length === 0 && (
                    <p className="mini-text text-gray text-center py-10">No columns defined for this table configuration.</p>
                  )}
                </div>

                {/* Code snippet preview */}
                <div className="p-12 border-ec rounded-8" style={{ background: "#0b0f19", borderLeft: "3.5px solid #7a1be7" }}>
                  <div className="flex justify-between items-center mb-6">
                    <p className="mini-text font-600" style={{ color: "#7a1be7" }}>React Code Component</p>
                    <button
                      onClick={() => handleCopyText(getTableCode(), "DataTable Component Copied!")}
                      className="bg-light-primary text-primary px-8 py-3 rounded-5 mini-text cursor-pointer font-600"
                      style={{ border: "none" }}
                    >
                      Copy
                    </button>
                  </div>
                  <pre className="p-8 rounded-5 mini-text text-light" style={{ overflowX: "auto", fontFamily: "monospace", color: "#e2e8f0", background: "#0f172a" }}>
                    {getTableCode()}
                  </pre>
                </div>
              </div>
            )}

            {/* 3. Field Config UI */}
            {activeConfigTab === "field" && (
              <div className="flex flex-column gap-16">
                <div className="flex justify-between items-center">
                  <h4 className="mid-text font-600 text-dark">Form Builder Fields</h4>
                  <button
                    onClick={handleAddFieldItem}
                    className="bg-light-primary text-primary px-10 py-5 rounded-5 mini-text font-600 cursor-pointer"
                    style={{ border: "none" }}
                  >
                    + Add Form Field
                  </button>
                </div>

                <div className="flex flex-column gap-10">
                  {fieldsList.map((field, idx) => (
                    <div
                      key={idx}
                      className="flex items-center gap-10 p-8 border-ec rounded-5"
                      style={{ background: "var(--bg-color)" }}
                    >
                      <div style={{ flex: 1.2 }}>
                        <Fields
                          type="input"
                          value={field.label}
                          placeholder="Label text"
                          onChange={(val) => handleFieldItemChange(idx, "label", val)}
                          border={true}
                        />
                      </div>
                      <div style={{ flex: 1 }}>
                        <Fields
                          type="input"
                          value={field.name}
                          placeholder="Name (key)"
                          onChange={(val) => handleFieldItemChange(idx, "name", val)}
                          border={true}
                        />
                      </div>
                      <div style={{ width: "135px" }}>
                        <Fields
                          type="select"
                          value={field.type}
                          options={[
                            { label: "Input Field", value: "input" },
                            { label: "Number Input", value: "number" },
                            { label: "Email Input", value: "email" },
                            { label: "Telephone Input", value: "tel" },
                            { label: "Password Field", value: "password" },
                            { label: "Datepicker", value: "datepicker" },
                            { label: "Colorpicker", value: "color" },
                            { label: "Textarea box", value: "textarea" },
                            { label: "Switch switch", value: "switch" }
                          ]}
                          onChange={(val) => handleFieldItemChange(idx, "type", val)}
                          border={true}
                        />
                      </div>
                      <button
                        onClick={() => handleDeleteFieldItem(idx)}
                        className="text-danger flex items-center justify-center p-4 cursor-pointer"
                        style={{ background: "none", border: "none" }}
                        title="Delete field"
                      >
                        🗑️
                      </button>
                    </div>
                  ))}
                  {fieldsList.length === 0 && (
                    <p className="mini-text text-gray text-center py-10">No fields added to this Form builder configuration.</p>
                  )}
                </div>

                {/* Code snippet preview */}
                <div className="p-12 border-ec rounded-8" style={{ background: "#0b0f19", borderLeft: "3.5px solid #7a1be7" }}>
                  <div className="flex justify-between items-center mb-6">
                    <p className="mini-text font-600" style={{ color: "#7a1be7" }}>React Code Component</p>
                    <button
                      onClick={() => handleCopyText(getFieldCode(), "FormBuilder Component Copied!")}
                      className="bg-light-primary text-primary px-8 py-3 rounded-5 mini-text cursor-pointer font-600"
                      style={{ border: "none" }}
                    >
                      Copy
                    </button>
                  </div>
                  <pre className="p-8 rounded-5 mini-text text-light" style={{ overflowX: "auto", fontFamily: "monospace", color: "#e2e8f0", background: "#0f172a" }}>
                    {getFieldCode()}
                  </pre>
                </div>
              </div>
            )}

          </div>
        </div>
      </main>
    </div>
  );
};

export default JsonEditorPage;
